**Goal:** Replace flat image-filter portrait mode with a depth-layered compositing system that produces expressive, 3D-feeling avatar animation via sixel rendering.

**Architecture:** 8 depth-ordered transparent PNG layers (background, hair, face, eyes, brows, nose, mouth, overlay) composited at startup into ~117 sixel-encoded frames. Parallax offsets per layer create 3D depth. At runtime, the avatar indexes into pre-baked frames for zero-CPU animation.

**Tech Stack:** Python 3.11+, Pillow (PIL), sixel encoder (existing), pytest

from PIL import Image, ImageEnhance


# GITS color grade: shift highlights toward cyan, midtones toward violet,
# shadows toward deep purple.
def _apply_gits_color_grade(img: Image.Image) -> Image.Image:
    """Apply Ghost in the Shell color grading to an RGB image."""
    pixels = img.load()
    w, h = img.size
    for y in range(h):
        for x in range(w):
            r, g, b = pixels[x, y]
            lum = (r + g + b) / 3.0
            if lum > 170:
                # Highlights: shift toward cyan
                r = int(r * 0.7)
                g = min(255, int(g * 1.1))
                b = min(255, int(b * 1.2))
            elif lum > 85:
                # Midtones: shift toward violet
                r = min(255, int(r * 1.05))
                g = int(g * 0.85)
                b = min(255, int(b * 1.15))
            else:
                # Shadows: shift toward deep purple
                r = min(255, int(r * 0.9 + 15))
                g = int(g * 0.7)
                b = min(255, int(b * 0.85 + 20))
            pixels[x, y] = (max(0, r), max(0, g), max(0, b))
    return img


# Map head angle keywords to parallax multipliers (-1.0 to 1.0)
_ANGLE_MULTIPLIERS: dict[str, float] = {
    "center": 0.0,
    "left": -1.0,
    "right": 1.0,
    "up": 0.0,
    "down": 0.0,
}


class LayerCompositor:
    """Composites depth-ordered PNG layers into final avatar frames.

    Args:
        assets_dir: Path to the layers asset directory.
        canvas_size: Pixel dimensions for the composite canvas.
    """

    def __init__(
        self,
        assets_dir: Path | None = None,
        canvas_size: tuple[int, int] = CANVAS_SIZE,
    ) -> None:
        self._assets_dir = assets_dir or ASSETS_DIR
        self._canvas_size = canvas_size
        self._loaded_layers: dict[str, dict[str, Image.Image]] = {}
        self._load_all_layers()

    def _load_all_layers(self) -> None:
        """Load all layer variant PNGs into memory."""
        for layer_name, layer_def in LAYER_DEFS.items():
            self._loaded_layers[layer_name] = {}
            for variant in layer_def["variants"]:
                fpath = self._assets_dir / variant["file"]
                if fpath.exists():
                    img = Image.open(fpath).convert("RGBA")
                    if img.size != self._canvas_size:
                        img = img.resize(self._canvas_size, Image.Resampling.LANCZOS)
                    self._loaded_layers[layer_name][variant["name"]] = img
                else:
                    log.warning("Missing layer asset: %s", fpath)

    def composite(
        self,
        combo: dict[str, str],
        head_angle: str = "center",
    ) -> Image.Image:
        """Composite a single frame from a layer combination dict.

        Args:
            combo: Maps layer name -> variant name.
            head_angle: "center", "left", "right", "up", "down" —
                controls parallax offset direction.

        Returns:
            Composited RGB image at canvas_size.
        """
        canvas = Image.new("RGB", self._canvas_size, (10, 10, 15))
        multiplier = _ANGLE_MULTIPLIERS.get(head_angle, 0.0)

        # Composite layers bottom-to-top (sorted by depth)
        for layer_name in LAYER_DEFS:
            variant_name = combo.get(layer_name)
            if variant_name is None:
                continue
            layer_img = self._loaded_layers.get(layer_name, {}).get(variant_name)
            if layer_img is None:
                continue

            # Apply parallax offset
            offset_px = int(PARALLAX_OFFSETS[layer_name] * multiplier)
            if offset_px != 0:
                shifted = Image.new("RGBA", self._canvas_size, (0, 0, 0, 0))
                shifted.paste(layer_img, (offset_px, 0))
                layer_img = shifted

            canvas.paste(layer_img, (0, 0), mask=layer_img)

        # Apply GITS color grade
        canvas = _apply_gits_color_grade(canvas)

        return canvas
```

import pickle


# Head angle inference: derive angle from the face variant name in a combo.
_FACE_TO_ANGLE: dict[str, str] = {
    "center": "center",
    "left15": "left",
    "right15": "right",
    "up10": "up",
    "down10": "down",
}

FRAME_RATES: dict[str, float] = {
    "idle": 0.8,
    "thinking": 0.15,
    "speaking": 0.1,
    "listening": 0.4,
    "error": 0.2,
}


class FrameAtlasBuilder:
    """Builds sixel frame atlases from layer combos and caches to disk.

    Args:
        assets_dir: Path to layer PNGs.
        cache_dir: Path for caching composited sixel frames.
        pixel_width: Target sixel frame width in pixels.
        pixel_height: Target sixel frame height in pixels.
        max_colors: Sixel palette size.
    """

    def __init__(
        self,
        assets_dir: Path | None = None,
        cache_dir: Path | None = None,
        pixel_width: int = 512,
        pixel_height: int = 512,
        max_colors: int = 128,
    ) -> None:
        self._assets_dir = assets_dir or ASSETS_DIR
        self._cache_dir = cache_dir or CACHE_DIR
        self._pixel_width = pixel_width
        self._pixel_height = pixel_height
        self._max_colors = max_colors

    def _cache_key(self) -> str:
        """Compute a cache key from assets checksum + render dimensions."""
        h = hashlib.sha256()
        h.update(f"{self._pixel_width}x{self._pixel_height}".encode())
        h.update(f"colors={self._max_colors}".encode())
        # Hash all layer files for invalidation
        for layer_name, layer_def in LAYER_DEFS.items():
            for variant in layer_def["variants"]:
                fpath = self._assets_dir / variant["file"]
                if fpath.exists():
                    h.update(fpath.read_bytes())
                else:
                    h.update(variant["file"].encode())
        return h.hexdigest()[:16]

    def _cache_path(self) -> Path:
        return self._cache_dir / self._cache_key() / "atlas.pkl"

    def _load_cache(self) -> dict[str, list[str]] | None:
        """Try to load cached frame atlas from disk."""
        path = self._cache_path()
        if path.exists():
            try:
                with open(path, "rb") as f:
                    data = pickle.load(f)
                log.info("Loaded frame atlas from cache: %s", path)
                return data
            except Exception as e:
                log.warning("Cache load failed: %s", e)
        return None

    def _save_cache(self, frames: dict[str, list[str]]) -> None:
        """Save frame atlas to disk cache."""
        path = self._cache_path()
        path.parent.mkdir(parents=True, exist_ok=True)
        with open(path, "wb") as f:
            pickle.dump(frames, f)
        log.info("Saved frame atlas to cache: %s", path)

    def build(self) -> tuple[dict[str, list[str]], dict[str, float]]:
        """Build or load the complete sixel frame atlas.

        Returns:
            (frames_dict, rates_dict) matching the standard frame set interface.
        """
        cached = self._load_cache()
        if cached is not None:
            return cached, FRAME_RATES

        from avatar.frames.sixel import encode_sixel

        compositor = LayerCompositor(
            assets_dir=self._assets_dir,
            canvas_size=CANVAS_SIZE,
        )

        frames: dict[str, list[str]] = {}

        for state, combos in STATE_FRAME_MAP.items():
            state_frames: list[str] = []
            for combo in combos:
                # Infer head angle from face variant
                face_variant = combo.get("face", "center")
                head_angle = _FACE_TO_ANGLE.get(face_variant, "center")

                # Composite the layers
                img = compositor.composite(combo, head_angle=head_angle)

                # Resize to target pixel dimensions
                if img.size != (self._pixel_width, self._pixel_height):
                    img = img.resize(
                        (self._pixel_width, self._pixel_height),
                        Image.Resampling.LANCZOS,
                    )

                # Encode to sixel
                sixel_str = encode_sixel(img, max_colors=self._max_colors)
                state_frames.append(sixel_str)

            frames[state] = state_frames
            log.info("Built %d frames for state '%s'", len(state_frames), state)

        self._save_cache(frames)
        return frames, FRAME_RATES
```

"""Generate layer assets for the layered 2.5D avatar system.

Procedural layers (backgrounds, overlays) are created with PIL.
Face/expression layers require AI generation (Phase 2).

Usage:
    python scripts/generate_layers.py [--output assets/layers] [--canvas-size 512]

from __future__ import annotations

import argparse
import math
import random
from pathlib import Path

from PIL import Image, ImageDraw, ImageFilter


def generate_backgrounds(output_dir: Path, canvas_size: tuple[int, int] = (512, 512)) -> None:
    """Generate 3 background layer PNGs: dim, pulse, error."""
    output_dir.mkdir(parents=True, exist_ok=True)
    w, h = canvas_size

    # bg_dim: dark gradient with subtle noise
    img = Image.new("RGBA", canvas_size, (10, 10, 15, 255))
    draw = ImageDraw.Draw(img)
    for y in range(h):
        # Subtle vertical gradient — slightly lighter at center
        dist = abs(y - h // 2) / (h // 2)
        brightness = int(15 + (1 - dist) * 8)
        for x in range(w):
            noise = random.randint(-3, 3)
            val = max(0, min(255, brightness + noise))
            draw.point((x, y), fill=(val, val, val + 5, 255))
    img.save(output_dir / "bg_dim.png")

    # bg_pulse: dim base with cyan radial glow at center
    img = Image.new("RGBA", canvas_size, (10, 10, 15, 255))
    draw = ImageDraw.Draw(img)
    cx, cy = w // 2, h // 2
    max_dist = math.sqrt(cx**2 + cy**2)
    for y in range(h):
        for x in range(w):
            dist = math.sqrt((x - cx)**2 + (y - cy)**2) / max_dist
            glow = max(0, int(40 * (1 - dist)))
            r = 10
            g = 10 + glow // 2
            b = 15 + glow
            draw.point((x, y), fill=(r, g, b, 255))
    img.save(output_dir / "bg_pulse.png")

    # bg_error: dark base with red vignette
    img = Image.new("RGBA", canvas_size, (15, 5, 5, 255))
    draw = ImageDraw.Draw(img)
    for y in range(h):
        for x in range(w):
            dist = math.sqrt((x - cx)**2 + (y - cy)**2) / max_dist
            glow = max(0, int(50 * (1 - dist)))
            r = 15 + glow
            g = 5
            b = 5
            draw.point((x, y), fill=(r, g, b, 255))
    img.save(output_dir / "bg_error.png")


def generate_overlays(output_dir: Path, canvas_size: tuple[int, int] = (512, 512)) -> None:
    """Generate 8 overlay effect PNGs as semi-transparent layers."""
    output_dir.mkdir(parents=True, exist_ok=True)
    w, h = canvas_size

    # scanline_light: horizontal lines at 30% opacity
    img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)
    for y in range(0, h, 4):
        draw.line([(0, y), (w, y)], fill=(0, 0, 0, 40))
    img.save(output_dir / "scanline_light.png")

    # scanline_heavy: horizontal lines at 60% opacity, denser
    img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)
    for y in range(0, h, 2):
        draw.line([(0, y), (w, y)], fill=(0, 0, 0, 80))
    img.save(output_dir / "scanline_heavy.png")

    # crt_bloom: bright center glow (cyan tint)
    img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)
    cx, cy = w // 2, h // 2
    max_dist = math.sqrt(cx**2 + cy**2)
    for y in range(h):
        for x in range(w):
            dist = math.sqrt((x - cx)**2 + (y - cy)**2) / max_dist
            alpha = max(0, int(30 * (1 - dist)))
            draw.point((x, y), fill=(100, 200, 255, alpha))

    draw = ImageDraw.Draw(img)
    for y in range(0, h, 6):
        draw.rectangle([0, y, w, y + 1], fill=(255, 0, 0, 25))

    img.save(output_dir / "chrom_aberr.png")

    # glitch_corrupt: random block artifacts
    img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)
    random.seed(42)  # Deterministic for consistency
    for _ in range(20):
        bx = random.randint(0, w - 50)
        by = random.randint(0, h - 10)
        bw = random.randint(20, 80)

    random.seed()
    img.save(output_dir / "noise_bands.png")

def generate_face_layers(
    reference: Image.Image,
    output_dir: Path,
    canvas_size: tuple[int, int] = (512, 512),

    """Generate face structure layers from a reference image.

    Creates face (5 angles), hair (3 positions), nose (3 positions)
    by applying affine transforms to the reference.

    # --- Face layers: perspective warp for head angle ---
    face_dir = output_dir / "face"
    face_dir.mkdir(parents=True, exist_ok=True)

    # Extract face region (center 60% of image)
    face_box = (int(w * 0.2), int(h * 0.1), int(w * 0.8), int(h * 0.85))
    face_region = ref.crop(face_box)

    # Center
    face_center = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
    face_center.paste(face_region, (face_box[0], face_box[1]))
    face_center.save(face_dir / "face_center.png")

    # Left 15 deg: shift face region left + slight horizontal squash
    for angle_name, x_offset, scale_x in [
        ("face_left15.png", -15, 0.95),
        ("face_right15.png", 15, 0.95),
    ]:
        shifted = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
        fw = int(face_region.width * scale_x)
        resized = face_region.resize((fw, face_region.height), Image.Resampling.LANCZOS)
        shifted.paste(resized, (face_box[0] + x_offset, face_box[1]))
        shifted.save(face_dir / angle_name)

    # Up/Down: vertical shift
    for angle_name, y_offset in [("face_up10.png", -8), ("face_down10.png", 8)]:
        shifted = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
        shifted.paste(face_region, (face_box[0], face_box[1] + y_offset))
        shifted.save(face_dir / angle_name)

    # --- Hair layers ---
    hair_dir = output_dir / "hair"
    hair_dir.mkdir(parents=True, exist_ok=True)

    # Extract top portion as "hair"
    hair_box = (0, 0, w, int(h * 0.4))
    hair_region = ref.crop(hair_box)

    for name, x_offset in [("hair_center.png", 0), ("hair_left.png", -8), ("hair_right.png", 8)]:
        img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
        img.paste(hair_region, (x_offset, 0))
        img.save(hair_dir / name)

    # --- Nose layers ---
    nose_dir = output_dir / "nose"
    nose_dir.mkdir(parents=True, exist_ok=True)

    # Small region at center of face
    nose_box = (int(w * 0.42), int(h * 0.45), int(w * 0.58), int(h * 0.6))
    nose_region = ref.crop(nose_box)

    for name, x_offset in [("nose_center.png", 0), ("nose_left.png", -3), ("nose_right.png", 3)]:
        img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
        img.paste(nose_region, (nose_box[0] + x_offset, nose_box[1]))
        img.save(nose_dir / name)


def generate_expression_layers(
    output_dir: Path,
    canvas_size: tuple[int, int] = (512, 512),
) -> None:
    """Generate expression layers: eyes, eyebrows, mouth.

    These are drawn geometrically in a cyberpunk style.
    Designed to be replaced with AI-generated variants later.
    """
    w, h = canvas_size

    # --- Eyes ---
    eyes_dir = output_dir / "eyes"
    eyes_dir.mkdir(parents=True, exist_ok=True)

    # Eye geometry constants
    eye_y = int(h * 0.38)  # Vertical center of eyes
    left_eye_x = int(w * 0.32)
    right_eye_x = int(w * 0.68)
    eye_w = int(w * 0.10)
    eye_h_open = int(h * 0.06)
    eye_h_half = int(h * 0.03)

    # Pupil offsets per direction
    pupil_offsets = {
        "center": (0, 0),
        "left": (-int(eye_w * 0.3), 0),
        "right": (int(eye_w * 0.3), 0),
        "up": (0, -int(eye_h_open * 0.3)),
        "down": (0, int(eye_h_open * 0.3)),
    }

    for direction in ["center", "left", "right", "up", "down"]:
        px, py = pupil_offsets[direction]
        for state in ["open", "half", "closed"]:
            img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
            draw = ImageDraw.Draw(img)

            if state == "closed":
                # Just thin lines
                for ex in [left_eye_x, right_eye_x]:
                    draw.line(
                        [(ex - eye_w, eye_y), (ex + eye_w, eye_y)],
                        fill=(0, 220, 200, 220), width=2,
                    )
            else:
                eh = eye_h_open if state == "open" else eye_h_half
                for ex in [left_eye_x, right_eye_x]:
                    # Eye outline (cyan)
                    draw.ellipse(
                        [ex - eye_w, eye_y - eh, ex + eye_w, eye_y + eh],
                        outline=(0, 220, 200, 220), width=2,
                    )
                    # Fill (dark)
                    draw.ellipse(
                        [ex - eye_w + 2, eye_y - eh + 2, ex + eye_w - 2, eye_y + eh - 2],
                        fill=(10, 10, 20, 200),
                    )
                    # Pupil (bright cyan)
                    pupil_r = max(3, eh // 2)
                    draw.ellipse(
                        [ex + px - pupil_r, eye_y + py - pupil_r,
                         ex + px + pupil_r, eye_y + py + pupil_r],
                        fill=(0, 255, 220, 255),
                    )
                    # Highlight dot
                    draw.ellipse(
                        [ex + px - pupil_r + 2, eye_y + py - pupil_r + 1,
                         ex + px - pupil_r + 5, eye_y + py - pupil_r + 4],
                        fill=(200, 255, 255, 200),
                    )

            img.save(eyes_dir / f"eyes_{direction}_{state}.png")

    # --- Eyebrows ---
    brows_dir = output_dir / "eyebrows"
    brows_dir.mkdir(parents=True, exist_ok=True)

    brow_y = eye_y - int(h * 0.06)
    brow_w = int(eye_w * 1.3)

    brow_configs = {
        "neutral": [(0, 0, 0, 0)],  # flat
        "raised": [(0, -6, 0, -6)],  # both up
        "furrowed": [(0, 4, 0, -2)],  # inner down, outer up
        "asymmetric": [(0, -5, 0, 0)],  # left raised only
    }

    for name, offsets in brow_configs.items():
        img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
        draw = ImageDraw.Draw(img)
        inner_l, outer_l, inner_r, outer_r = offsets[0]

        # Left brow
        draw.line(
            [(left_eye_x - brow_w, brow_y + outer_l),
             (left_eye_x + brow_w, brow_y + inner_l)],
            fill=(0, 200, 180, 200), width=3,
        )
        # Right brow
        draw.line(
            [(right_eye_x - brow_w, brow_y + inner_r),
             (right_eye_x + brow_w, brow_y + outer_r)],
            fill=(0, 200, 180, 200), width=3,
        )
        img.save(brows_dir / f"brows_{name}.png")

    # --- Mouth ---
    mouth_dir = output_dir / "mouth"
    mouth_dir.mkdir(parents=True, exist_ok=True)

    mouth_x = w // 2
    mouth_y = int(h * 0.62)
    mouth_w = int(w * 0.10)

    mouth_configs = {
        "closed": {"h": 1, "fill": (0, 180, 160, 180)},
        "slight": {"h": int(h * 0.015), "fill": (0, 180, 160, 200)},
        "open": {"h": int(h * 0.035), "fill": (10, 10, 20, 220)},
        "wide": {"h": int(h * 0.055), "fill": (10, 10, 20, 240)},
        "smile": {"h": int(h * 0.02), "fill": (0, 200, 180, 200), "curve": True},
        "glitch": {"h": int(h * 0.04), "fill": (200, 0, 50, 200), "noise": True},
    }

    for name, cfg in mouth_configs.items():
        img = Image.new("RGBA", canvas_size, (0, 0, 0, 0))
        draw = ImageDraw.Draw(img)
        mh = cfg["h"]

        if name == "closed":
            draw.line(
                [(mouth_x - mouth_w, mouth_y), (mouth_x + mouth_w, mouth_y)],
                fill=cfg["fill"], width=2,
            )
        elif cfg.get("curve"):
            # Smile: arc
            draw.arc(
                [mouth_x - mouth_w, mouth_y - mh, mouth_x + mouth_w, mouth_y + mh],
                start=0, end=180, fill=cfg["fill"], width=2,
            )
        elif cfg.get("noise"):
            # Glitch mouth: jagged lines
            random.seed(77)
            for i in range(5):
                y_off = random.randint(-mh, mh)
                draw.line(
                    [(mouth_x - mouth_w + random.randint(-5, 5), mouth_y + y_off),
                     (mouth_x + mouth_w + random.randint(-5, 5), mouth_y + y_off)],
                    fill=cfg["fill"], width=random.randint(1, 3),
                )
            random.seed()
        else:
            # Open mouth shapes: ellipse
            draw.ellipse(
                [mouth_x - mouth_w, mouth_y - mh, mouth_x + mouth_w, mouth_y + mh],
                outline=cfg["fill"], fill=(10, 10, 20, 200), width=2,
            )

        img.save(mouth_dir / f"mouth_{name}.png")
